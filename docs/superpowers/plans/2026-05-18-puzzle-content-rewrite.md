# Pictok Puzzle Content Rewrite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Produce a 60-puzzle `puzzles.json` that passes the spec's authoring bar (no pure label compounds), wire it into the iOS app bundle, and confirm the Today view loads a real puzzle at runtime.

**Architecture:** Content authoring with phased user-review checkpoints (Phase A calibration sample → Phase B per-category batches → Phase C assembly + mechanical validation → Phase D date assignment + bundle wiring). Mechanical pieces (validation, date assignment) are short Python scripts in `scripts/`; the app code itself does not change.

**Tech Stack:** JSON content, Python 3 (for validation + date scripts), xcodegen (project regen), Xcode build toolchain, iPhone 17 / iOS 26.5 Simulator.

**Spec:** `docs/superpowers/specs/2026-05-18-puzzle-content-rewrite-design.md` — read this before authoring any puzzle. The authoring rules (banned vs. allowed) are non-negotiable; the workflow phases are the source of truth for what each task delivers.

## File structure

| Path | Purpose | Created in |
|------|---------|------------|
| `puzzles-wip.json`        | Work-in-progress build-up of puzzles across Phase B. Lives at repo root so it's easy to inspect. Discarded after Phase C assembly merges it into final form. | Phase A → Phase B |

- [ ] **Step 3: Write `puzzles-wip.json` with the 10 entries**

Schema per entry (matches `Puzzle.swift`):
```json
{
  "id": "puzzle-001",
  "date": "",
  "emoji": "<2-4 emojis, no whitespace>",
  "answer": "<UPPERCASE, spaces preserved, no punctuation>",
  "category": "<Movie|Song|Book|Brand|Celeb>",
  "subcategory": "<short hint string, never reveals answer>",
  "difficulty": "<medium|hard>"

```

IDs: `puzzle-001` through `puzzle-010`, zero-padded to 3 digits, sequential in author order.

- [ ] **Step 4: Self-review each entry against the instant-read test**

For each row, write down (in your head) the decode steps a player must do. If the count of decode steps is zero, the puzzle is banned — replace it. Minimum decode work for Medium = 1 abstraction step or 1 phonetic step. Minimum for Hard = 2 steps OR a single non-obvious phonetic mapping.

- [ ] **Step 5: Deliver to user as a markdown table for review**

Render the 10 entries as a markdown table with columns: `#`, `category`, `difficulty`, `emoji`, `answer`, `decode explanation`. The `decode explanation` column shows the work the player must do (this is for review only, never goes in the JSON). Example row:

```
| 001 | Brand | hard | 💧🚽 | WATERLOO | water + loo (British slang for toilet) → "Waterloo" |
```

Wait for user accept/reject/fix per row.

- [ ] **Step 6: Iterate until user approves all 10**

For each rejected row, write a replacement and re-deliver only the changed rows. Goal: zero rejections in the next batch (Phase B).

- [ ] **Step 7: Flag structural concerns from authoring difficulty**

If during Steps 2-3 you found that Book or Celeb candidates ran dry before hitting 2 quality puzzles, surface this **explicitly** to the user before Phase B starts. Per the spec, the user picks the fallback (drop the category and rebalance to four, or keep the category and overweight Movie/Brand).

- [ ] **Step 2: Write `scripts/validate-puzzles.py`**

```python
#!/usr/bin/env python3
"""Validate the puzzle JSON file: schema, duplicates, distribution."""
import json, sys, collections

REQUIRED_FIELDS = {"id", "date", "emoji", "answer", "category", "subcategory", "difficulty"}
VALID_CATEGORIES = {"Movie", "Song", "Book", "Brand", "Celeb"}
VALID_DIFFICULTIES = {"medium", "hard"}

def main(path: str) -> int:
    with open(path) as f:
        puzzles = json.load(f)

    errors: list[str] = []

    if not isinstance(puzzles, list):
        return _fail([f"Top-level JSON must be a list, got {type(puzzles).__name__}"])

    # Schema check
    for i, p in enumerate(puzzles):
        missing = REQUIRED_FIELDS - p.keys()
        if missing:
            errors.append(f"#{i} ({p.get('id','?')}): missing fields {sorted(missing)}")
        if p.get("category") not in VALID_CATEGORIES:
            errors.append(f"#{i} ({p.get('id','?')}): invalid category {p.get('category')!r}")
        if p.get("difficulty") not in VALID_DIFFICULTIES:
            errors.append(f"#{i} ({p.get('id','?')}): invalid difficulty {p.get('difficulty')!r}")
        if not p.get("answer", "").isupper() and p.get("answer"):
            errors.append(f"#{i} ({p.get('id','?')}): answer must be UPPERCASE: {p.get('answer')!r}")
        if not p.get("emoji"):
            errors.append(f"#{i} ({p.get('id','?')}): emoji empty")

    # Duplicate check
    ids = [p["id"] for p in puzzles if "id" in p]
    for dup, n in collections.Counter(ids).items():
        if n > 1:
            errors.append(f"duplicate id: {dup} (×{n})")

    answers = [p["answer"] for p in puzzles if "answer" in p]
    for dup, n in collections.Counter(answers).items():
        if n > 1:
            errors.append(f"duplicate answer: {dup!r} (×{n})")

    emojis = [p["emoji"] for p in puzzles if "emoji" in p]
    for dup, n in collections.Counter(emojis).items():
        if n > 1:
            errors.append(f"duplicate emoji pattern: {dup!r} (×{n})")

    # Distribution check
    cat_counts = collections.Counter(p["category"] for p in puzzles if "category" in p)
    diff_counts = collections.Counter(p["difficulty"] for p in puzzles if "difficulty" in p)
    print(f"Total: {len(puzzles)} puzzles")
    print(f"Categories: {dict(cat_counts)}")
    print(f"Difficulties: {dict(diff_counts)}")

    if errors:
        return _fail(errors)
    print("OK: all checks passed.")
    return 0

def _fail(errs: list[str]) -> int:
    print("FAILED:", file=sys.stderr)
    for e in errs:
        print(f"  - {e}", file=sys.stderr)
    return 1

if __name__ == "__main__":
    sys.exit(main(sys.argv[1] if len(sys.argv) > 1 else "puzzles.json"))
```

```python
#!/usr/bin/env python3
"""Populate the `date` field of each puzzle sequentially from a start date.

Usage:
  python3 scripts/assign-dates.py <path-to-puzzles.json> <YYYY-MM-DD>

Mutates the file in place.
"""
import json, sys, datetime as dt

def main(path: str, start_iso: str) -> int:
    start = dt.date.fromisoformat(start_iso)
    with open(path) as f:
        puzzles = json.load(f)
    for i, p in enumerate(puzzles):
        p["date"] = (start + dt.timedelta(days=i)).isoformat()
    with open(path, "w") as f:
        json.dump(puzzles, f, ensure_ascii=False, indent=2)
        f.write("\n")
    print(f"Wrote {len(puzzles)} dates: {puzzles[0]['date']} … {puzzles[-1]['date']}")
    return 0

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print(__doc__, file=sys.stderr)
        sys.exit(2)
    sys.exit(main(sys.argv[1], sys.argv[2]))
```
